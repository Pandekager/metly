  <action>
Create `frontend/server/api/refund_return_analysis.get.ts`:

Follow the pattern from `frontend/server/api/revenue_leak_analysis.get.ts`:

1. Import auth store to get JWT token
2. Use useRequestHeaders to forward authorization header
3. Fetch from backend API `/api/refund_return_analysis`
4. Return the response data directly

Example structure:
```typescript
import { useAuthStore } from '~/stores/auth'

export default defineEventHandler(async (event) => {
  const authStore = useAuthStore()
  const token = authStore.token
  
  const headers = useRequestHeaders(['authorization'])
  
  const data = await $fetch<RefundReturnResponse>(
    `${process.env.API_BASE || 'http://127.0.0.1:8000/api'}/refund_return_analysis`,
    {
      headers: {
        ...headers,
        authorization: `Bearer ${token}`,
      },
    }
  )
  
  return data
})
```
